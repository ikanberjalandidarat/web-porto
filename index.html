class SimState:
    def __init__(self, sim):
        self.sim = sim

    def transition(self, new_state):
        self.sim.state.on_exit()
        self.sim.state = new_state
        new_state.on_enter()

    def handle_action(self, action):
        pass

class Standing(SimState):
    def on_enter(self):
        self.sim.current_position = 'Standing'
        print(f"Sim is now standing and feeling {self.sim.emotion}.")
        self.sim.update_graph()

    def handle_action(self, action):
        if action == "sit_down":
            self.transition(Sitting(self.sim))
        elif action == "walk":
            self.transition(Walking(self.sim))
        elif action == "run":
            self.transition(Running(self.sim))

class DQN(nn.Module):
    def __init__(self, state_size, action_size):
        super(DQN, self).__init__()
        self.fc1 = nn.Linear(state_size, 128)
        self.fc2 = nn.Linear(128, 128)
        self.fc3 = nn.Linear(128, action_size)
        
    def forward(self, x):
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        return self.fc3(x)
